$('.finish-slider__track').slick({

              slidesToShow: 2,

          {

            settings: {
              slidesToShow: 1,
            }
          },

      $('.finish-slider__prev').click(() => {
        $('.finish-slider__track').slick('slickPrev')
      });

        $('.finish-slider__track').slick('slickNext')
      });
